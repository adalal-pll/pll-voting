    Google Sheets Setup Guide

    Step 1 — Create the Google Sheet

    Create a new Google Sheet at sheets.google.com. Name it "PLL All-Star Voting 2026". You don't need to create any tabs — the script does that automatically.

    Step 2 — Open Apps Script

    In your Sheet, click Extensions → Apps Script. Delete any existing code and paste the entire script below:

    // PLL All-Star Voting — Apps Script Backend // Paste this entire script, then deploy as a Web App const SHEET_NAME = 'PLL All-Star Voting 2025'; // must match your sheet name function doGet(e) { var action = e.parameter.action; if (action === 'getRoster') return jsonResp(getRoster()); if (action === 'getVotes') return jsonResp(getVotes()); if (action === 'getBallots') return jsonResp(getBallots()); if (action === 'getHasVoted') return jsonResp(getHasVoted()); if (action === 'ping') return jsonResp({status:'ok', ts: new Date().toISOString()}); return jsonResp({error:'unknown action'}); } function doPost(e) { var data = JSON.parse(e.postData.contents); var action = data.action; if (action === 'uploadRoster') return jsonResp(uploadRoster(data)); if (action === 'submitVote') return jsonResp(submitVote(data)); if (action === 'deleteBallot') return jsonResp(deleteBallot(data)); if (action === 'togglePlayer') return jsonResp(togglePlayer(data)); if (action === 'addPlayer') return jsonResp(addPlayer(data)); if (action === 'removePlayer') return jsonResp(removePlayer(data)); return jsonResp({error:'unknown action'}); } // ── HELPERS ────────────────────────────────────────── function ss() { return SpreadsheetApp.getActiveSpreadsheet(); } function getOrCreateSheet(name, headers) { var s = ss().getSheetByName(name); if (!s) { s = ss().insertSheet(name); if (headers) s.appendRow(headers); } return s; } function sheetToObjects(sheet) { var data = sheet.getDataRange().getValues(); if (data.length < 2) return []; var headers = data[0]; return data.slice(1).map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); } function jsonResp(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ── ROSTER ──────────────────────────────────────────── function getRoster() { var sheet = ss().getSheetByName('Roster'); if (!sheet) return {players: []}; return {players: sheetToObjects(sheet)}; } function uploadRoster(data) { var sheet = getOrCreateSheet('Roster', ['id','firstName','lastName','age','rawTeam','team','division','active','manual']); // Clear existing CSV-sourced rows (keep manual=TRUE rows) var rows = sheet.getDataRange().getValues(); var toKeep = [rows[0]]; // keep header for (var i = 1; i < rows.length; i++) { if (String(rows[i][8]).toUpperCase() === 'TRUE') toKeep.push(rows[i]); // manual players } sheet.clearContents(); if (toKeep.length) sheet.getRange(1, 1, toKeep.length, toKeep[0].length).setValues(toKeep); // Append new players data.players.forEach(function(p) { sheet.appendRow([p.id, p.firstName, p.lastName, p.age, p.rawTeam, p.team, p.division, true, false]); }); return {status:'ok', count: data.players.length}; } function togglePlayer(data) { var sheet = ss().getSheetByName('Roster'); if (!sheet) return {error:'no roster'}; var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(data.id)) { var newVal = data.active; sheet.getRange(i + 1, 8).setValue(newVal); return {status:'ok'}; } } return {error:'player not found'}; } function addPlayer(data) { var sheet = getOrCreateSheet('Roster', ['id','firstName','lastName','age','rawTeam','team','division','active','manual']); sheet.appendRow([data.id, data.firstName, data.lastName, data.age, data.rawTeam, data.team, data.division, true, true]); return {status:'ok'}; } function removePlayer(data) { var sheet = ss().getSheetByName('Roster'); if (!sheet) return {error:'no roster'}; var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(data.id)) { sheet.deleteRow(i + 1); return {status:'ok'}; } } return {error:'not found'}; } // ── VOTES ───────────────────────────────────────────── function getVotes() { var sheet = ss().getSheetByName('Votes'); if (!sheet) return {votes: {}}; var rows = sheetToObjects(sheet); var votes = {}; rows.forEach(function(r) { votes[r.key] = { name: r.name, team: r.team, division: r.division, age: r.age, count: Number(r.count) || 0, voters: r.voters ? String(r.voters).split('|').filter(Boolean) : [] }; }); return {votes: votes}; } function getHasVoted() { var sheet = ss().getSheetByName('HasVoted'); if (!sheet) return {hasVoted: {}}; var rows = sheetToObjects(sheet); var hv = {}; rows.forEach(function(r) { hv[r.key] = r.ballotId; }); return {hasVoted: hv}; } function getBallots() { var sheet = ss().getSheetByName('Ballots'); if (!sheet) return {ballots: []}; return {ballots: sheetToObjects(sheet)}; } function submitVote(data) { // 1. Write to Votes tab (tally) var vSheet = getOrCreateSheet('Votes', ['key','name','team','division','age','count','voters']); var vRows = vSheet.getDataRange().getValues(); data.players.forEach(function(p) { var key = p.key; var found = false; for (var i = 1; i < vRows.length; i++) { if (vRows[i][0] === key) { var newCount = (Number(vRows[i][5]) || 0) + 1; var newVoters = vRows[i][6] ? vRows[i][6] + '|' + data.team : data.team; vSheet.getRange(i+1, 6).setValue(newCount); vSheet.getRange(i+1, 7).setValue(newVoters); vRows[i][5] = newCount; vRows[i][6] = newVoters; found = true; break; } } if (!found) { var newRow = [key, p.name, p.team, p.division, data.ageGroup, 1, data.team]; vSheet.appendRow(newRow); vRows.push(newRow); } }); // 2. Write to Ballots tab var bSheet = getOrCreateSheet('Ballots', ['id','voter','voterName','team','age','ts','players']); bSheet.appendRow([ data.ballotId, data.voter, data.voterName, data.team, data.ageGroup, data.ts, JSON.stringify(data.players) ]); // 3. Write to HasVoted tab var hvSheet = getOrCreateSheet('HasVoted', ['key','email','age','ballotId','ts']); hvSheet.appendRow([ data.voter + '__' + data.ageGroup, data.voter, data.ageGroup, data.ballotId, data.ts ]); return {status:'ok'}; } function deleteBallot(data) { var ballotId = data.ballotId; // Find ballot var bSheet = ss().getSheetByName('Ballots'); if (!bSheet) return {error:'no ballots sheet'}; var bRows = bSheet.getDataRange().getValues(); var ballot = null; var bRowIdx = -1; for (var i = 1; i < bRows.length; i++) { if (String(bRows[i][0]) === String(ballotId)) { ballot = { voter: bRows[i][1], age: bRows[i][4], team: bRows[i][3], players: JSON.parse(bRows[i][6] || '[]') }; bRowIdx = i + 1; break; } } if (!ballot) return {error:'ballot not found'}; // Decrement vote tallies var vSheet = ss().getSheetByName('Votes'); if (vSheet) { var vRows = vSheet.getDataRange().getValues(); ballot.players.forEach(function(p) { for (var i = 1; i < vRows.length; i++) { if (vRows[i][0] === p.key) { var newCount = Math.max(0, (Number(vRows[i][5]) || 0) - 1); var voters = String(vRows[i][6] || '').split('|'); var idx = voters.indexOf(ballot.team); if (idx >= 0) voters.splice(idx, 1); vSheet.getRange(i+1, 6).setValue(newCount); vSheet.getRange(i+1, 7).setValue(voters.filter(Boolean).join('|')); if (newCount <= 0) vSheet.deleteRow(i + 1); break; } } }); } // Remove HasVoted entry var hvSheet = ss().getSheetByName('HasVoted'); if (hvSheet) { var hvKey = ballot.voter + '__' + ballot.age; var hvRows = hvSheet.getDataRange().getValues(); for (var i = 1; i < hvRows.length; i++) { if (String(hvRows[i][0]) === hvKey) { hvSheet.deleteRow(i + 1); break; } } } // Delete ballot row bSheet.deleteRow(bRowIdx); return {status:'ok'}; }

    Step 3 — Deploy

    Click Deploy → New Deployment. Set type to Web App. Set "Execute as" to Me and "Who has access" to Anyone. Click Deploy and copy the URL.

    Step 4 — Paste URL

    Paste the Web App URL into the "Google Sheets Connection" field in the Admin Dashboard. Click "Save & Test Connection". A green success message means everything is working.

    Step 5 — Upload roster

    Upload your CSV in the Roster section. The roster writes directly to the Sheet's "Roster" tab and is immediately available to all voters.

    Every new season: Upload a new CSV. Optionally clear the Votes, Ballots, and HasVoted tabs in the Sheet to reset voting.
